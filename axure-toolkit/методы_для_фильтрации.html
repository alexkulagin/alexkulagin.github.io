Метод clearFilters

/* отфильтрует строки с именем admin_1 и типом super */

var filterA = { condition: { name: 'admin_1', type: 'super' } };


/* отфильтрует строки с активностью 0 */

var filterB = { condition: { active: 0 } };


/* применяет первый фильтр */

controller.applyFilter(filterA);


/* применяет второй фильтр в качестве исключения */

controller.applyFilter(filterB, true);


/* обновляет представление */

controller.render();

Применяет новый фильтр.
